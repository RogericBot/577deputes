\subsection{DetecciĂłn de enmiendas casi idĂ©nticas}\label{sec:clusters-es}

\paragraph{(a) DefiniciĂłn.} Se busca agrupar las enmiendas cuyo texto dispositivo
es \og{}esencialmente el mismo\fg{}, compartan o no autor. Para una enmienda $a$ se
construye $S_a$, el conjunto de los $k$-\emph{shingles} de palabras (secuencias de
$k$ palabras consecutivas) del texto normalizado, cada uno troceado a 64~bits. Dos
enmiendas $a,a'$ se consideran casi idĂ©nticas si $J(S_a,S_{a'})\ge\theta$. Un
\emph{clĂşster} es una componente conexa (en el sentido de la uniĂłn de esos pares)
de tamaĂ±o $\ge 2$.

\paragraph{(b) Algoritmo y parĂˇmetros.}
\begin{enumerate}[label=\arabic*.]
  \item \emph{NormalizaciĂłn}: supresiĂłn de etiquetas HTML; descomposiciĂłn Unicode
        (NFKD) y supresiĂłn de diacrĂ­ticos; paso a minĂşsculas; supresiĂłn de todo
        carĂˇcter fuera de \texttt{[a-z0-9 ]}; normalizaciĂłn de espacios; truncado a
        $L_{\max}=12\,000$ caracteres.
  \item \emph{Shingling}: $k = 5$; $S_a=\{h(w_i\cdots w_{i+k-1})\}$ con $h$ un
        hash BLAKE2b de 64~bits. Enmienda descartada si $|S_a| < s_{\min}=8$.
  \item \emph{Firma MinHash}: $K=64$ funciones hash $g_j(x)=x\oplus\sigma_j$
        ($\sigma_j$ semillas deterministas), firma $\mathrm{sig}_j(a)=\min_{x\in S_a} g_j(x)$.
  \item \emph{Bloqueo LSH}: la firma se divide en $b=16$ bandas de $r=4$
        componentes; $a$ y $a'$ son \emph{candidatos} si coinciden en al menos una
        banda. La probabilidad de que un par de similitud real $s$ sea candidato es
        $P(s)=1-(1-s^{\,r})^{b}$ (curva en S, umbral $\approx (1/b)^{1/r}\approx 0{,}55$).
  \item \emph{VerificaciĂłn}: para cada par candidato, cĂˇlculo del Jaccard exacto
        sobre $S_a$, $S_{a'}$; par retenido si $J\ge\theta=0{,}80$.
  \item \emph{Agrupamiento}: union-find sobre los pares retenidos; clĂşsteres de
        tamaĂ±o $\ge 2$.
\end{enumerate}

$\approx 9\,460$ clĂşsteres que agrupan $28\,124$ enmiendas (la base de producciĂłn
registra $28\,055$ â€” la diferencia de $0{,}25~\%$ proviene de una salvaguarda de
tamaĂ±o de \emph{bucket} introducida en el script de verificaciĂłn, y constituye
nuestro control de reproducibilidad).

\subsection{Disciplina de voto (cohesiĂłn individual)}\label{sec:discipline-es}

\paragraph{(a) DefiniciĂłn.} Sea $G_d$ el grupo del diputado $d$. Sea $V_d$ el
conjunto de las votaciones donde $d$ ha emitido un voto $p_d(v)\in\{\text{a favor},\text{en contra},\text{abstenciĂłn}\}$

\[

\mathrm{disciplina}(d)=\frac{\mathrm{alineados}(d)}{\mathrm{emitidos}(d)}.

\subsection{Matriz de cohesiĂłn entre grupos}\label{sec:matrice-es}

\paragraph{(a) DefiniciĂłn.} Para dos grupos $G_1,G_2$, sea $V_{12}$ el conjunto de
las votaciones donde ambos tienen una posiciĂłn mayoritaria definida. La cohesiĂłn es
\[

\]

\paragraph{(b) Algoritmo y parĂˇmetros.} NingĂşn parĂˇmetro libre; una auto-uniĂłn
sobre la tabla de posiciones mayoritarias.

\paragraph{(c) Sensibilidad.} Una elecciĂłn discutible: una \og{}abstenciĂłn $=$
abstenciĂłn\fg{} cuenta como acuerdo. Recalculando la matriz sin las votaciones donde
ambos grupos se abstienen, la diferencia absoluta media es de $0{,}21$~puntos y el
mĂˇximo de $0{,}83$~puntos. El efecto es, pues, \emph{despreciable}.

\paragraph{(d) Literatura.} Es una matriz de \emph{acuerdo} entre bloques en el
sentido de Hix-Noury-Roland~\cite{hnr2007}; no proporciona ningĂşn orden
unidimensional (para ello, vĂ©anse los mĂ©todos espaciales citados en
Â§\ref{sec:discipline-es}).

\paragraph{(e) LĂ­mites.} CohesiĂłn calculada sobre \emph{posiciones de grupo}, no
sobre votos individuales: un grupo dividido $51/49$ y un grupo unĂˇnime cuentan
igual. Dos \og{}en contra\fg{} por motivos opuestos son indistinguibles de un
acuerdo de fondo.

\subsection{Absentismo estratĂ©gico}\label{sec:absence-es}

\paragraph{(a) DefiniciĂłn.} Entre las votaciones con $\text{emitidos}(v)\ge \param{T_v}=200$,

\emph{consensual} si $>\param{C}=0{,}50$, en otro caso sin clasificar. Para un

\emph{proporciĂłn} de votaciones divisivas (resp.\ consensuales) en las que $d$ ha

muestra si $d$ estĂˇ en ejercicio y $\mathrm{tot}_{div}(d)\ge \param{T_{div}}=30$.

\paragraph{(b)--(c) ParĂˇmetros y sensibilidad.} Cuatro umbrales

que la mediana del \emph{gap} estĂˇ cerca de $0$ â€” la mayorĂ­a de los diputados tienen

\paragraph{(d)--(e) Literatura y lĂ­mites.} La idea se relaciona con la literatura
